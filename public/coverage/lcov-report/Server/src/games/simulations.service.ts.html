import { Game, Control, Organisation, Settings } from '../_helpers/db';

/**
 * Represents the games service.
 */
export default class SimulationsService {
  /**
   * Simulate a turn occuring within a game.
   *
   * This method also handles game logic such as detecting the end of a game.
   *
   * @category Simulations
   *
   * @param gameId  The ID of the game to simulate a turn for.
   * @returns  The simulated events for the turn.
   */
  async simulateTurn(gameId: Types.ObjectId): Promise<EventClass[]> {
    return await this._simulateEventsForTurn(gameId);
  }
 
  /**
   * Simulate the events of a given turn.
   *
   * @category Simulations
   *
   * @param gameId  The ID of the game to run the simulations for.
   * @returns  The list of events resulting from the simulation(s).
   */
  protected async _simulateEventsForTurn(gameId: Types.ObjectId): Promise<EventClass[]> {
    const {
      id,
      source,
      currentTurn,
      minNumOfEvents,
      maxNumOfEvents,
      minCostPerEvent,
      maxCostPerEvent,
    } = await gamesService.get(gameId);
 
    const numOfEvents = Number(Math.floor(Math.random() * (maxNumOfEvents - minNumOfEvents) + minNumOfEvents));
    console.debug(`Simulating turn ${currentTurn} of game ${id}: ${numOfEvents} events`);
 
    const eventsForTurn: EventClass[] = [];
 
    for (let i = 1; i <= numOfEvents; i++) {
      const threatActor: string = await this._generateThreatActor();
      const asset: Asset = await this._generateAsset();
      const securityAreas: SecurityAreaClass[] = await this._generateSecurityAreas(source);
      const cost: number = this._generateCost(minCostPerEvent, maxCostPerEvent);
      const name: string = this._generateName(cost);
      const description: string = this._generateDescription({ asset, threatActor, securityAreas });
 
      const event: EventClass = await eventsService.create({
        game: gameId,
        turn: currentTurn,
        name,
        description,
        cost,
        threatActor,
        asset: asset.name,
        securityAreas: securityAreas.map((securityArea) => securityArea.id),
      });
      eventsForTurn.push(event);
    }
 
    return eventsForTurn;
  }
 
  private _generateName(cost?: number): string {
    if (cost) {
      if (cost > 2500.0) {
        return 'A catastrophic incident';
      } else if (cost > 1500.0) {
        return 'A serious incident';
      } else if (cost > 800.0) {
        return 'An incident';
      } else return 'A minor incident';
    } else return 'An incident';
  }
 
  // TODO: should this be handled on the front-end?
  private _generateDescription({
    cost,
    securityAreas,
    asset,
    threatActor,
  }: {
    cost?: number;
    securityAreas?: SecurityAreaClass[];
    asset?: Asset;
    threatActor?: string;
  }): string {
    let description = '';
    const assetDesc = asset ? `Your ${asset.name} was` : 'You were';
    const threatActorDesc = threatActor ? `a ${threatActor}` : `an unknown actor`;
 
    if (securityAreas) {
      description += `${assetDesc} attacked by ${threatActorDesc}.<br><br>`;
      description += 'The attack was classified into the following area(s):<br><ul class="table__list">';
      securityAreas.forEach((securityArea) => {
        description += `<li class="table__list__item">${securityArea.number}. ${securityArea.name}</li>`;
      });
      description += '</ul>';
      return description;
    } else return "<em>Something</em> bad has happened, but you don't know anything more than that.";
  }
 
  private _generateCost(min: number, max: number): number {
    return Number((Math.random() * (max - min + 1) + min).toFixed(2));
  }
 
  private async _generateThreatActor(): Promise<string> {
    const { value: threatActors }: { value: string[] } = await Settings.findOne({ key: 'threatActors' });
    const { value: threatActorProbabilities }: { value: number[] } = await Settings.findOne({
      key: 'threatActorProbabilities',
    });
 
    const lastIndex = threatActorProbabilities.length - 1;
    const rand = Math.random();
    let sum = 0.0;
    let threat: string;
 
    if (threatActors.length !== threatActorProbabilities.length) throw 'Array length mismatch';
 
    threatActors.forEach((threatActor, idx) => {
      sum += threatActorProbabilities[idx];
      if (rand <= sum && !threat) {
        threat = threatActor;
      }
    });
 
    return threat;
  }
 
  private async _generateAsset(): Promise<Asset> {
    const { value: assets }: { value: Asset[] } = await Settings.findOne({
      key: 'assets',
    });
 
    const equalProb = 1.0 / assets.length;
    const rand = Math.random();
 
    let sum = 0.0;
    let thisAsset: Asset;
 
    assets.forEach((asset) => {
      sum += equalProb;
 
      if (rand <= sum && !thisAsset) {
        thisAsset = asset;
      }
    });
 
    return thisAsset;
  }

    const allSecurityAreas: SecurityAreaClass[] = await securityAreasService.getAllBySource(source);
 
    const numOfSecurityAreas: number = Math.floor(Math.random() * (3 - 1 + 1) + 1);

    // TODO: This should not be assigned randomly.
    for (let i = 1; i <= numOfSecurityAreas; i++) {
      const randIdx = Math.floor(Math.random() * (allSecurityAreas.length - 1 - 0) + 0);
 
      if (allSecurityAreas[randIdx]) eventSecurityAreas.push(allSecurityAreas[randIdx]);
      else {

        console.debug(`rand = ${randIdx}`);
        console.debug(`result = ${allSecurityAreas[randIdx]}`);

    }
 
    return eventSecurityAreas.sort((a, b) => Number(a.number) - Number(b.number));
  }
